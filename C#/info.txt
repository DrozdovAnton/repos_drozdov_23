5.1. Increment & Decrement.
	5.1.1.
	int a = 0;
	int b = 0;
 
	Console.Writeline(a++);
	// после выполнения строчки выше переменная a = 1
	Console.Writeline(b--);
	// после выполнения строчки выше переменная b = -1

	output: 0
		0

	5.1.2.
	int a = 0;
	int b = 0;
 
	Console.Writeline(++a); // переменная а сначала увеличивается на единицу, далее вывод в консоль
	Console.Writeline(--b); // переменная b сначала уменьшается на единицу, далее вывод в консоль

	output: 1
		-1
